Consider the following C program:
```
#define LARGE_INT 1000000
extern int unknown_int();

int main() {
    int x,y;
    x = -50;
    y = unknown_int();
    if (!(-1000 < y && y < LARGE_INT)) return 0;
    while (x < 0) {
	x = x + y;
	y++;
    }
    //@ assert(y > 0);
    return 0;
}
```
Output the loop invariants for the loop in the program above.
